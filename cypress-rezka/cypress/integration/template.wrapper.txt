import { getStream } from '../get-stream.wrapper';

describe('%TEMPLATE_TEST_NAME%', () => {
	it('get-stream', () => {
		getStream({
			url: '%TEMPLATE_URL%',
			callback: (result) => {
				const name = result.url.split('/').pop();
				cy.writeFile('rezka-files/' + name, result);
			},
		});
	});
});
